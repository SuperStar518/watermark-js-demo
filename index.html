watermark.js

Center

                    
    watermark(['./img/{YOUR_IMAGE}.jpg'])
    .image(text.center('KINDY', '60px serif', 'green', 0.5))
    .then(function (img) {
        document.getElementById('center-placed').appendChild(img);
    });

Lower Left

                    
    watermark(['./img/{YOUR_IMAGE}.jpg'])
    .image(text.lowerLeft('watermark.js', '60px Josefin Slab', 'red', 0.5))
    .then(function (img) {
        document.getElementById('lower-left').appendChild(img);
    });
